import socket
import time

ip = '172.20.10.3'

sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

while (True):
    try:
        sock.connect((ip,port))
        break
    except ConnectionRefusedError:
        print("waiting for server...")
        sock.close()
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        time.sleep(1)

import socket
import time
import sys

port = 9876
ip = '172.20.10.3'

sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

while True:
    try:
        sock.connect((ip,port))
        break
    except ConnectionRefusedError:
        print("waiting for server...")
        sock.close()
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        time.sleep(1)

while True:
    text = input(">")
    encoded_text = text.encode()
    try:
        sock.send(encoded_text)
    except BrokenPipeError:
        print("remote site disconnected")
        sys.exit()

import socket
import sys

port = 9876

sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
sock.bind( ('', port) )
sock.listen(1)

while True:
    newsock, addr = sock.accept()

    while(True):
        encoded_text = newsock.recv(1024)
        if len(encoded_text) == 0:
            print("remote site disconnected")
            newsock.close()
            break
        text = encoded_text.decode()
        print(">>", text, "<<")

blocking cell

in client, text = input(">") blocks the server, the program is waiting for input from user.

in server, encoded_text = newsock.recv(1024) blocks the server, waiting for the input from client.
