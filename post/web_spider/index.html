    #得到指定某个url的网页内容
    def askURL(url):
        head = {
            "User-Agent": "Mozilla / 5.0(Windows NT 6.1; ) AppleWebKit / 537.36(KHTML, like Gecko) Chrome / 81..4044 .129Safari / 537.36"
        }
        request = urllib.request.Request(url, headers=head)
        html = ""
        try:
            response = urllib.request.urlopen(request)
            html = response.read().decode("utf-8")
            print(html)
        except urllib.error.URLError as e:
            #hasattr函数检测e是否有code属性，如果是true，就把e.code打印下来
            if hasattr(e, "code"):
                print(e.code)
            if hasattr(e, "reason"):
                print(e.reason)
        return html

注意在复制headers相关信息时，不可以有多余的空格，否则可能导致418

  • hasattr函数：hasattr(object, name)，name是属性名，如果对象有该属性返回True，否则返回False

#影片详情页面的规则
findlink = re.compile(r'<a href="(.*?)">')  # 创建正则表达式对象，表示字符串的模式
#<a href="https://movie.douban.com/subject/1292052/">

#影片图片链接
findImgSrc = re.compile(r'<img.*src="(.*?)"', re.S) #re.s忽略换行符
#<img alt="肖申克的救赎" class="" src="https://img3.doubanio.com/view/photo/s_ratio_poster/public/p480747492.jpg" width="100"/>

#影片片名
findTitle = re.compile(r'<span class="title">(.*)</span>')
#<span class="title">肖申克的救赎</span>

#影片评分
findrating = re.compile(r'<span class="rating_num" property="v:average">(.*)</span>')
#<span class="rating_num" property="v:average">9.7</span>

#评价人数
findjudge = re.compile(r'<span>(\d*)人评价</span>')
#<span>2073489人评价</span>

#找到概况
findinq = re.compile(r'<span class="inq">(.*)</span>')
#<span class="inq">希望让人自由。</span>

#找到影片相关内容
findbd = re.compile(r'<p class="">(.*)</p>', re.S)
#p class="">
                            #导演: 弗兰克·德拉邦特 Frank Darabont   主演: 蒂姆·罗宾斯 Tim Robbins /...<br/>
                            #1994 / 美国 / 犯罪 剧情
                        #</p>

用正则表达式来表示相应的信息

  • 前面的r表示忽视特殊符号,防止对网址里的/进行错误解析

  • 找到链接前后固定的开头和结尾，中间各种字符用.*表示（有零个或一个各种字符）

        # 解析数据
        soup = BeautifulSoup(html, "html.parser")
        for item in soup.find_all('div',class_="item"): #查找符合要求的字符串，形成列表
            #print(item)   #测试：查看一个item就是一个电影的全部信息
            data = []   #保存一部电影的全部信息
            item = str(item)

            #保存一部电影的详情链接
            link = re.findall(findlink, item)[0]     #通过正则表达式查找指定的字符串
            data.append(link)

            imgSrc = re.findall(findImgSrc, item)[0]
            data.append(imgSrc)

            title = re.findall(findTitle, item)  #片名可能同时有中文和外文名
            if (len(title)) == 2:
                ctitle = title[0]
                data.append(ctitle)
                otitle = title[1].replace("/", '') #去掉前面无关符号
                data.append(otitle)
            else:
                data.append(title[0])
                data.append(' ') #为外文名留空

            rating = re.findall(findrating, item)[0]
            data.append(rating)            #添加评分

            judgeNum = re.findall(findjudge, item)[0]
            data.append(judgeNum)         #添加评价人数

            inq = re.findall(findinq, item)
            if len(inq) != 0:     #概述
                inq = inq[0].replace("。",",")     #去掉中文句号
                data.append(inq)
            else:
                data.append(" ")

            bd = re.findall(findbd, item)[0]
            bd = re.sub('<br(\s+)?/>(\s+)?', " ", bd)   #去掉<br/>
            bd = re.sub('/'," ",bd)     #替换/
            data.append(bd.strip())    #去掉前后空格

            datalist.append(data)     #把一部处理好的电影信息放入datalist里

    return datalist

提取了html中的有用信息并且放到data列表里，最后一起放到datalist里

  • 如果一些信息可能缺失（比如外文名）就需要在列表里用'‘把位置补上

  • 去掉无关元素可以用otitle = title[1].replace("/", '') #去掉前面无关符号或者bd = re.sub('<br(\s+)?/>(\s+)?', " ", bd) #去掉<br/> \s表示空白,包括空格、换行、tab缩进等所有的空白

def saveData(datalist, savepath):
    book = xlwt.Workbook(encoding="utf-8", style_compression=0)  # 创建workbook对象,一个文件
    sheet = book.add_sheet('豆瓣电影top250',cell_overwrite_ok=True)  # 创建工作表

    col = ('电影详情链接', "图片链接", "中文名", "外文名", "评分", "评价数", "概况", "相关信息")
    for i in range(0,8):
        sheet.write(0, i, col[i])  #写列名

    for i in range(0,250):
        print("第%d条"%(i+1))   #打印写入进度
        data = datalist[i]   #获取一个电影的整个信息
        for j in range(0,8):
            sheet.write(i+1, j, data[j])   #一个电影的信息在同一行上

    print('爬取完毕')
    book.save('豆瓣.xls')     #保存


    sheet.write(0, 0, 'hello')  # 写入数据，第一个参数表示行，第二个参数表示列，第三个参数是内容
    book.save(savepath)  # 保存
