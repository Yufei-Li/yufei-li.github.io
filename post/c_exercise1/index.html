#include <stdio.h>
#include <string.h>

// a number list from 11 to 19
char num_another_list[9][10] = {"eleven","twelve","thirteen","fourteen","fifteen","sixteen","seventeen","eighteen","nineteen"};
//a number list of 20, 30, 40...
char num_last_list[8][8] = {"twenty", "thirty", "forty", "fifty", "sixty", "seventy", "eighty", "ninety"};
//a number list from 0 to 10
char num_list[12][6] = {"zero", "one", "two","three", "four", "five", "six", "seven", "eight", "nine", "ten"};


char* read_num(int num){
		// the function that can read a integer, return a string(char pointer)
    if (num >= 0 && num <= 10){
        return num_list[num];
    }else if (num >= 11 && num <= 19){
        return num_another_list[num - 11];
    }else if (num % 10 == 0){
        int a = num / 10;
        char *result = num_last_list[a - 2];
        return result;
    }else if (num >= 20 && num <= 99){
      // if num >= 20 && num <= 99, we need to combine two strings
        int a, b;
        a = num / 10;
        char *a_string = num_last_list[a - 2];
        b = num % 10;
        char *b_string = num_list[b];
        strcat(a_string, "-");
        strcat(a_string, b_string);
        return a_string;
    }else{
        return "I don't know how to read this";
    }
}

int main()
{

    int num;
    printf("please enter a number between 0 and 999:");
    scanf("%d", &num);

    printf("%s", read_num(num));
    return 0;
}

  • When use array for the strings, the number in the second (eg.num_another_list[9][10] ) should plus 1, give the extra space for “\0”, otherwise c don’t know where to stop.

  • the syntax to create a new string array char num_another_list[9][10] = {"eleven","twelve","thirteen","fourteen","fifteen","sixteen","seventeen","eighteen","nineteen"};

  • char *a_string = num_last_list[a - 2]; the name of string is a point, so it need a * before it

reverse a integer

Write a program to determine if a long int (type long) entered via the keyboard is a palindrome (i.e., represents the same value when reversed, for example 123454321).

  • To check whether a integer is palindrome, firstly we should inverse the integer, then check whether int == reverse(int)

  • To get the reversed integer, we need to get the number of digit of this integer

#include <stdio.h>
#include <math.h>

int len(long a){
    int num = 0;
    while (a > 0){
        a = a / 10;
        num ++;
    }
    return num;
}

long reverse_num(long a){
    long reverse = 0;
    int length = len(a);
    for (int i = 0; i < length; i++){
        int num = a % 10;
        a = a / 10;
        reverse += num * pow(10, (length - i - 1));
    }
    return reverse;
}

#include <stdio.h>
#include <stdlib.h>

int valid_num();
int is_prime(long num);

int is_prime(long num){
    // whether the number is prime number
    int prime = 1;
    if (num == 1 || num == 0){
        prime = 0;
        return prime;
    }else if (num == 2){
        return prime;
    }else{
        for(int i = 2; i <= num - 1; i++)
        {
            if( num % i == 0)
            {
                prime=0;
                return prime;
            }
        }
    }

    return prime;
}


int valid_num(){
        char input = getchar();
        if (input == '\n'){
            return 1;
        } else if (input == ' '){
            valid_num();
        } else{
            return 0;
        }
}


void function(){
    long num;
    printf("please enter a valid number:");
    scanf("%ld", &num);

    if(valid_num()){
        if (is_prime(labs(num))){
            printf("this is a prime number\n");
        }else{
            printf("this is not a prime number\n");
        }
    }else{
        fflush(stdin);
        function();
    }
}

int main()
{
    function();

    return 0;
}
