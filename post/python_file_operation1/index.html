def test_nums_texts():
    file = open("nums.txt", "wt") # write as text 1000000 1000001 1000002 1000003 1000004 ...
    for i in range(1000000, 1000010):
        file.write(str(i) + "\n")
    file.close()

def text_nums_bytes():
    file = open("nums.bin", "wb")
    for i in range(1000000, 1000010):
        file.write(i.to_bytes(4, byteorder="little"))
    file.close()

    with open("nums.bin", "rb") as file2:
        file2.seek(8) # set file after 8 bytes
        b = file2.read(4)
        i = int.from_bytes(b, byteorder="little")
        print(i)
        print(file2.tell()) # current position in file
