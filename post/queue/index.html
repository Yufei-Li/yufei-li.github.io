class Node():
    def __init__(self, value):
        self.value = value
        self.next = None

    def append(self, node):
        if self.next is not None:
            raise (ValueError("next node is not None"))
        self.next = node

def test_node():
    n1 = Node(1)
    assert (n1.value == 1)
    assert (n1.next == None)

    n2 = Node(2)
    n1.append(n2)
    assert (n1.next is not None)

    n3 = Node(3)
    assert (n2.next is None)
    n2.append(n3)
    assert (n2.next is not None)

    node = n1
    count = 1
    while node.next is not None:
        node = node.next
        count += 1

    assert (node == n3)
    assert (count == 3)

class Node():
    def __init__(self, value):
        self.value = value
        self.next = None

    def append(self, node):
        if self.next is not None:
            raise (ValueError("next node is not None"))
        self.next = node

class Queue():
    def __init__(self):
        # create a new queue
        self.head = None
        self.tail = None

    def add(self, value):
        # add a new value to the end of the queue
        n = Node(value)
        if self.head is None:
            self.head = n
            self.tail = n
        else:
            self.tail.append(n)
            self.tail = n

    def pop(self):
        # remove the first queued element, and return its value
        if self.head is None:
            raise ValueError("attempt to pop from an empty queue")
        value = self.head.value
        self.head = self.head.next

        if self.head is None:
            # if we remove the last item
            self.tail = None
        return value

    def is_empty(self):
        return self.head is None
